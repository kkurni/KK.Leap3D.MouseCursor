144  float angleTo(const Vector& other) const {
145  float denom = this->magnitudeSquared() * other.magnitudeSquared();
146  if (denom <= 0.0f) {
147  return 0.0f;
148  }
149  return std::acos(this->dot(other) / std::sqrt(denom));

493  void setRotation(const Vector& _axis, float angleRadians) {
494  const Vector axis = _axis.normalized();
495  const float s = std::sin(angleRadians);
496  const float c = std::cos(angleRadians);
497  const float C = (1-c);
498 
499  xBasis = Vector(axis[0]*axis[0]*C + c, axis[0]*axis[1]*C - axis[2]*s, axis[0]*axis[2]*C + axis[1]*s);
500  yBasis = Vector(axis[1]*axis[0]*C + axis[2]*s, axis[1]*axis[1]*C + c, axis[1]*axis[2]*C - axis[0]*s);
501  zBasis = Vector(axis[2]*axis[0]*C - axis[1]*s, axis[2]*axis[1]*C + axis[0]*s, axis[2]*axis[2]*C + c);
502  }
503 
511  Vector transformPoint(const Vector& in) const {
512  return xBasis*in.x + yBasis*in.y + zBasis*in.z + origin;
513  }

520  Vector transformDirection(const Vector& in) const {
521  return xBasis*in.x + yBasis*in.y + zBasis*in.z;
522  }

531  Matrix rigidInverse() const {
532  Matrix rotInverse = Matrix(Vector(xBasis[0], yBasis[0], zBasis[0]),
533  Vector(xBasis[1], yBasis[1], zBasis[1]),
534  Vector(xBasis[2], yBasis[2], zBasis[2]));
535  rotInverse.origin = rotInverse.transformDirection( -origin );
536  return rotInverse;
537  }
